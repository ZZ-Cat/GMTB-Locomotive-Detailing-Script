                E:constraintBreak( "weld", W )
                E:parentTo( W )
                
            }

            noCollide( FrontBogey, LocomotiveBody )
            FrontBogey:propPhysicalMaterial( "friction_00" )
            
        }

        else {

            FrontBogey = CE[ 1, entity ]
